var findMissing = function (list) { 

    var arr = [];

    for(var i = 0; i < list.length-1; i ++){
        //得到相邻元素的差，并添加到数组中；
        arr.push(list[i+1]-list[i]);
    }
    for(var j = 0; j < arr.length; j ++){

        if(arr[j] !== arr[j+1]){

            //比较一下两个差的大小，选择那个绝对值比较小的差作为等差数列中的差；

            var bad = Math.abs(arr[j])<Math.abs(arr[j+1])?arr[j]:arr[j+1];

            //判断bad的值，如果bad的值等于arr中靠后的元素，那个索引值index = j;否则index = j+1；  

            var index = bad === arr[j+1]?j:j+1;

            return list[index] + bad；
        }
    }

 }
