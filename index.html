var findMissing = function (list) { 

    var arr = [];

    for(var i = 0; i < list.length-1; i ++){
        //得到相邻元素的差，并添加到数组中；
        arr.push(list[i+1]-list[i]);
    }
    for(var j = 0; j < arr.length; j ++){

        if(arr[j] !== arr[j+1]){

            //比较一下两个差的大小，选择那个绝对值比较小的差作为等差数列中的差；

            var bad = Math.abs(arr[j])<Math.abs(arr[j+1])?arr[j]:arr[j+1];

            //判断bad的值，如果bad的值等于arr中靠后的元素，那个索引值index = j;否则index = j+1；  

            var index = bad === arr[j+1]?j:j+1;

            return list[index] + bad；
        }
    }

要求：

  • 一个字符串中，每个单词中都包含一个数字，数字的大小决定最后这个单词的所在位置
  • 如果字符串是空的，返回空字符串

Example：

  • “is2 Thi1s T4est 3a” => return “Thi1s is2 3a T4est”

function order(words){

  //将字符串转化为数组

  var arr = words.split(' ');

      var num = arr.map(function(item,index){

function order(words) {
  return words.split(" ").sort((a, b) => a.replace(/[^\d]/g, "") - b.replace(/[^\d]/g, "")).join(" ")
}

sort()方法：

1. arrayObject.sort(sortby)；参数sortby可选。规定排序顺序。必须是函数。
2. 若 a 小于 b，在排序后的数组中 a 应该出现在 b 之前，则返回一个小于 0 的值。
3. 若 a 等于 b，则返回 0。
4. a 大于 b，则返回一个大于 0 的值。

说明：

字符串 stringObject 的 replace() 方法执行的是查找并替换的操作。它将在 stringObject 中查找与 regexp 相匹配的子字符串，然后用 replacement 来替换这些子串。如果 regexp 具有全局标志 g，那么 replace() 方法将替换所有匹配的子串。否则，它只替换第一个匹配子串。
